108  *
109  * The given typename will be the type that the resulting data

118  {
119  JSONPair kvp = keyValuePairs[i];
120  std::string kvpKey = kvp.getKey();
121  if (strcmp(kvpKey.c_str(), key) == 0)
122  {
123  return (T)kvp.getValue();
124  }
125  }
126  return 0;
127  }

188  {
189  JSONValue value;
190 
191  //Parsing Bools
192  if (strcmp(string, "true") == 0)
193  {
194  value = true;
195  }
196  else if (strcmp(string, "false") == 0)
197  {
198  value = false;
199  }
200  //Parsing strings
201  else if (string[0] == '\"')

242  inline static JSONObject* parseObject(const char* string)
243  {
244  int index = 0;
245  size_t stringLength = strlen(string);
246  char c = string[index];
247 
248  JSONObject* object = new JSONObject;
249 
250  bool parsing = true;
251 
252  while (parsing)
253  {
254  //We've hit a member; determine how long until the next member and then parse it
255  if (c == '\"')
256  {
257  int startIndex = index;
258  int memberLength = 0;
259  char m = c;
260  char targetChar = ',';
261 
262  int nestedObjects = 0;
263  int nestedArrays = 0;
264 
265  while (m != targetChar || nestedObjects > 0 || nestedArrays > 0)
266  {
267  //If we've read past our string length we can just break
268  if (index + memberLength >= stringLength)
269  {
270  parsing = false;
271  break;
272  }
273 
274  m = string[startIndex + memberLength];
275 
276  if (m == '}' && nestedObjects == 0)
277  break;
278 
279  if (m == '{')
280  nestedObjects++;
281  else if (m == '}' && nestedObjects > 0)
282  nestedObjects--;
283 
284  if (m == '[')
285  nestedArrays++;
286  else if (m == ']' && nestedArrays > 0)
287  nestedArrays--;
288 
289  memberLength++;
290  }
291 
292  //If we get a member that's less than 5 chars (by accident) just increment the index counter and keep going
293  //5 because the least number of chars for a member is 5
294  //Ex. "a":1
295  if (memberLength < 5)
296  {
297  index++;
298  continue;
299  }
300 
301  char* memberString = new char[memberLength + 1];
302  memcpy(memberString, string + startIndex, memberLength);
303 
304  //We MAY want to trim off the last character
305  if (memberString[memberLength - 1] == ','
306  || (!strstr(memberString, "{") && memberString[memberLength - 1] == '}'))
307  memberString[memberLength - 1] = '\0';
308  //Regardless the string needs a null terminator
309  else
310  memberString[memberLength] = '\0';
311 
312  JSONPair member = parseMember(std::string(memberString));
313 
314  //If the key is never set then there was a problem
315  if (member.getKey() != "")
316  object->addPair(member);
317  else
318  std::cout << "Error parsing member in object" << std::endl;
319 
320  delete[] memberString;
321 
322  index += (memberLength - 1);
323  }
324 
325  c = string[index++];
326 
327  //If we've read past our string length we can just break
328  if (index >= stringLength)
329  parsing = false;
330  }
331 
332  return object;
333  }
